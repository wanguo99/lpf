   ```c
   uint16_t prl_calc_crc16_update(uint16_t crc, const uint8_t *data, size_t len) {
       for (size_t i = 0; i < len; i++) {
           crc = crc16_table[(crc ^ data[i]) & 0xFF] ^ (crc >> 8);
       }
       return crc;
   }
   ```

2. **重构验证函数**（1天）
   ```c
   // core/prl/src/prl_common.c
   bool prl_verify_packet_crc(const uint8_t *packet, size_t packet_len) {
       // 分段计算CRC，避免拷贝
       uint16_t crc = prl_calc_crc16(packet, offsetof(prl_header_t, crc16));
       uint16_t zero = 0;
       crc = prl_calc_crc16_update(crc, (uint8_t*)&zero, sizeof(zero));
       crc = prl_calc_crc16_update(crc, 
           packet + offsetof(prl_header_t, crc16) + sizeof(uint16_t),
           packet_len - offsetof(prl_header_t, crc16) - sizeof(uint16_t));
       
       prl_header_t *header = (prl_header_t *)packet;
       return (crc == header->crc16);
   }
   ```

   ```c
   // core/acl/src/acl_api.c
   static const acl_config_table_t *g_acl_table = NULL;
   static osal_rwlock_t *g_acl_rwlock = NULL;
   
   int32_t ACL_Init(void) {
       int32_t ret = OSAL_RWLockCreate(&g_acl_rwlock);
       if (ret != OSAL_SUCCESS) {
           LOG_ERROR("ACL", "Failed to create rwlock");
           return ret;
       }
       return OSAL_SUCCESS;
   }
   
   int32_t ACL_Deinit(void) {
       OSAL_RWLockDelete(g_acl_rwlock);
       g_acl_rwlock = NULL;
       g_acl_table = NULL;
       return OSAL_SUCCESS;
   }
   ```

2. **保护所有访问**（1天）
   ```c
   int32_t ACL_RegisterTable(const acl_config_table_t *table) {
       OSAL_RWLockWriteLock(g_acl_rwlock);
       g_acl_table = table;
       OSAL_RWLockWriteUnlock(g_acl_rwlock);
       return OSAL_SUCCESS;
   }
   
   const acl_tc_config_t* ACL_GetTcConfig(uint32_t function_id) {
       OSAL_RWLockReadLock(g_acl_rwlock);
       const acl_tc_config_t *config = NULL;
       if (g_acl_table && function_id < g_acl_table->tc_count) {
           config = &g_acl_table->tc_table[function_id];
       }
       OSAL_RWLockReadUnlock(g_acl_rwlock);
       return config;
   }
   ```

   ```c
   int32_t OSAL_FileOpen(const char *path, int32_t flags, osal_fd_t *fd) {
       int posix_flags = 0;
       // 转换flags
       if (flags & OSAL_O_RDONLY) posix_flags |= O_RDONLY;
       if (flags & OSAL_O_WRONLY) posix_flags |= O_WRONLY;
       if (flags & OSAL_O_RDWR) posix_flags |= O_RDWR;
       
       int file_fd = open(path, posix_flags);
       if (file_fd < 0) {
           return (errno == ENOENT) ? OSAL_ERR_NOT_FOUND : OSAL_ERR_GENERIC;
       }
       
       *fd = (osal_fd_t)(intptr_t)file_fd;
       return OSAL_SUCCESS;
   }
   ```
